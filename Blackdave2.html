// Add keyboard event listeners document.addEventListener('keydown', function(event) { if (event.code === 'Space') { // Check if HIT button is visible if (document.getElementById('hit-button').offsetParent !== null) { document.getElementById('hit-button').click(); } // Check if NEW ROUND button is visible else if (document.getElementById('new-round-button').offsetParent !== null) { document.getElementById('new-round-button').click(); } // Check if DEAL button is visible else if (document.getElementById('place-bet-button').offsetParent !== null) { document.getElementById('place-bet-button').click(); } } if (event.code === 'Enter') { // Check if STAND button is visible if (document.getElementById('stand-button').offsetParent !== null) { document.getElementById('stand-button').click(); } } event.preventDefault(); // Prevent default browser behavior });
